      /**
     *realclass.java
     */
      package practice;

import java.lang.*;
import java.util.*;

public class realclass implements Comparator<realclass>,Comparable<realclass>{

    private int index;

    private String foo;

    //空值构造
    realclass(){}
    //带参构造
    realclass(int index,String foo){
        this.index=index;
        this.foo=foo;
    }

    public int getIndex() {
        return index;
    }

    public void setIndex(int index) {
        this.index = index;
    }

    public String getFoo() {
        return foo;
    }

    public void setFoo(String foo) {
        this.foo = foo;
    }

    public int compareTo(realclass real){
        return (this.foo).compareTo(real.foo);
    }

    @Override
    public int compare(realclass o1, realclass o2) {
        // TODO Auto-generated method stub
        return o1.getIndex()-o2.getIndex();
    }
}
